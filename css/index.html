    Various way to specify color in css

    element{
  /* Color Name */
  color:red;

  /* RGB (Red Green Blue) */
  color:rgb(255,0,0);

  /* HEX (Hexa Decimal) */
  color:#ff0000;

  /*HSL (Hue Saturation Lightness*/
  color:hsl(0, 100%, 50%);

  /*RGBA (Red Green Blue Alpha*/
  color:rgba(255,0,0,1.0);
}

    2. ID Selectors

}

    3. Class Selectors

    .blue {
  color: blue;
}

    4. * Selectors

    * {
  font-weight: 400;
}

    5. Descentdant Selectors

    body ul li a {
  color:white;
}

ol li a {
  color:white;
}

  font-size: 18px;
}

}

    6. Adjacent Selectors

    h1 + a {
  background:orange;
}

h2 + ul {
  background: green;
}

h3 + ol {
  background: blue;
}

li+a{
  list-style: none;
}

    7. Attribute Selectors

}

}

}

li:nth-of-type(1) {
  list-style: none;
}

}
